#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h> // لإضافة الطابع الزمني في السجل
#include "cJSON.h"
#include "config.h"

// --- بداية التعديلات ---

// تعريفات الألوان لسهولة الاستخدام
#define KRED "\x1B[31m"
#define KGRN "\x1B[32m"
#define KYEL "\x1B[33m"
#define KBLU "\x1B[34m"
#define KCYN "\x1B[36m"
#define KWHT "\x1B[37m"
#define KRESET "\x1B[0m"

/**
 * @brief تسجل التغييرات في ملف sniper-config.log
 * * @param home المسار الرئيسي للمستخدم لتحديد مكان ملف السجل.
 * @param action نوع الإجراء (SET, DELETE).
 * @param category الفئة.
 * @param key المفتاح.
 * @param value القيمة (تكون NULL في حالة الحذف).
 */
void log_change(const char *home, const char *action, const char *category, const char *key, const char *value) {
    char log_filepath[512];
    snprintf(log_filepath, sizeof(log_filepath), "%s/.config/sniper-config.log", home);

    FILE *log_file = fopen(log_filepath, "a");
    if (!log_file) {
        return; // لا تفعل شيئاً إذا لم نتمكن من فتح ملف السجل
    }

    time_t now = time(NULL);
    char time_str[20];
    strftime(time_str, sizeof(time_str), "%Y-%m-%d %H:%M:%S", localtime(&now));

    if (value) {
        fprintf(log_file, "[%s] %s: category='%s' key='%s' value='%s'\n", time_str, action, category, key, value);
    } else {
        fprintf(log_file, "[%s] %s: category='%s' key='%s'\n", time_str, action, category, key);
    }

    fclose(log_file);
}

// --- نهاية التعديلات ---

char* read_file(const char *filename) {
    FILE *file = fopen(filename, "r");
    if (!file) {
        // تلوين الخطأ
        fprintf(stderr, KRED "✖ Error:" KWHT " Configuration file not found at %s. Please create it first.\n" KRESET, filename);
        return NULL;
    }
    // ... باقي الكود لم يتغير ...
    fseek(file, 0, SEEK_END);
    long length = ftell(file);
    rewind(file);
    char *content = malloc(length + 1);
    if (!content) return NULL;
    fread(content, 1, length, file);
    content[length] = '\0';
    fclose(file);
    return content;
}

int write_file(const char *filename, const char *data) {
    FILE *file = fopen(filename, "w");
    if (!file) {
        fprintf(stderr, KRED "✖ Error:" KWHT " Could not write to file %s.\n" KRESET, filename);
        return 1;
    }
    fputs(data, file);
    fclose(file);
    return 0;
}

// تم تعديل الدالة لتقبل 'home' لتمريره لدالة السجل
int set_value(const char *filepath, const char *home, const char *category, const char *key, const char *value) {
    char *data = read_file(filepath);
    if (!data) return 1;

    cJSON *json = cJSON_Parse(data);
    if (!json) {
        fprintf(stderr, KRED "✖ Error:" KWHT " Failed to parse JSON. Check the file format.\n" KRESET);
        free(data);
        return 1;
    }

    cJSON *cat = cJSON_GetObjectItem(json, category);
    if (!cat) {
        cat = cJSON_CreateObject();
        cJSON_AddItemToObject(json, category, cat);
    }

    cJSON_ReplaceItemInObject(cat, key, cJSON_CreateString(value));
    char *out = cJSON_Print(json);

    int result = write_file(filepath, out);
    if (result == 0) {
        // تسجيل التغيير عند النجاح فقط
        log_change(home, "SET", category, key, value);
    }
    
    free(out);
    cJSON_Delete(json);
    free(data);

    return result;
}

int get_value(const char *filepath, const char *category, const char *key) {
    char *data = read_file(filepath);
    if (!data) return 1;

    cJSON *json = cJSON_Parse(data);
    if (!json) {
        fprintf(stderr, KRED "✖ Error:" KWHT " Failed to parse JSON. Check the file format.\n" KRESET);
        free(data);
        return 1;
    }

    cJSON *cat = cJSON_GetObjectItem(json, category);
    if (!cat) {
        printf(KYEL "⚠ Warning:" KWHT " Category '%s' not found.\n" KRESET, category);
    } else {
        cJSON *item = cJSON_GetObjectItem(cat, key);
        if (item && cJSON_IsString(item)) {
            // طباعة القيمة باللون الأزرق
            printf(KBLU "%s\n" KRESET, item->valuestring);
        } else {
            printf(KYEL "⚠ Warning:" KWHT " Key '%s' not found in category '%s'.\n" KRESET, key, category);
        }
    }

    cJSON_Delete(json);
    free(data);
    return 0;
}

// تم تعديل الدالة لتقبل 'home' لتمريره لدالة السجل
int delete_value(const char *filepath, const char *home, const char *category, const char *key) {
    char *data = read_file(filepath);
    if (!data) return 1;

    cJSON *json = cJSON_Parse(data);
    if (!json) {
        fprintf(stderr, KRED "✖ Error:" KWHT " Failed to parse JSON. Check the file format.\n" KRESET);
        free(data);
        return 1;
    }

    cJSON *cat = cJSON_GetObjectItem(json, category);
    if (cat) {
        if (cJSON_HasObjectItem(cat, key)) {
            cJSON_DeleteItemFromObject(cat, key);
        } else {
            printf(KYEL "⚠ Warning:" KWHT " Key '%s' not found, nothing to delete.\n" KRESET, key);
            cJSON_Delete(json);
            free(data);
            return 1; // يعتبر فشلاً لأنه لم يتم الحذف
        }
    } else {
        printf(KYEL "⚠ Warning:" KWHT " Category '%s' not found, nothing to delete.\n" KRESET, category);
        cJSON_Delete(json);
        free(data);
        return 1;
    }

    char *out = cJSON_Print(json);
    int result = write_file(filepath, out);

    if (result == 0) {
        // تسجيل التغيير عند النجاح فقط
        log_change(home, "DELETE", category, key, NULL);
    }

    free(out);
    cJSON_Delete(json);
    free(data);

    return result;
}

// دالة المساعدة الملونة والمحسنة
void print_help(void) {
    printf(KCYN "Sniper Config Manager" KRESET " - " KWHT "A simple tool to manage JSON configuration.\n\n" KRESET);
    printf(KYEL "USAGE:\n" KRESET);
    printf(KWHT "  configer " KGRN "<command>" KWHT " [" KYEL "category" KWHT "] [" KYEL "key" KWHT "] [" KYEL "value" KWHT "]\n\n");
    
    printf(KYEL "COMMANDS:\n" KRESET);
    printf(KGRN "  set    " KYEL "<category> <key> <value>" KWHT "    Set or update a configuration value.\n" KRESET);
    printf(KGRN "  get    " KYEL "<category> <key>" KWHT "            Retrieve a specific value.\n" KRESET);
    printf(KGRN "  delete " KYEL "<category> <key>" KWHT "            Delete a key-value pair.\n" KRESET);
    printf(KGRN "  help   " KWHT "                                Show this help message.\n\n");
    
    printf(KYEL "EXAMPLE:\n" KRESET);
    printf(KWHT "  configer set user prompt_text \"Hello Sniper\"\n" KRESET);
    printf(KWHT "  configer get user prompt_text\n" KRESET);
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "config.h"

// أضفنا تعريفات الألوان هنا للوصول السريع
#define KGRN "\x1B[32m"
#define KWHT "\x1B[37m"
#define KRESET "\x1B[0m"

int main(int argc, char *argv[]) {
    const char *home = getenv("HOME");
    if (!home) {
        fprintf(stderr, "HOME environment variable not found.\n");
        return 1;
    }

    char filepath[512];
    snprintf(filepath, sizeof(filepath), "%s/.config/sniper-config.json", home);

    if (argc < 2) {
        print_help();
        return 1;
    }

    if (strcmp(argv[1], "help") == 0) {
        print_help();
        return 0;
    }

    // --- بداية التعديلات ---

    if (strcmp(argv[1], "set") == 0 && argc == 5) {
        // تم تمرير متغير 'home' لإنشاء سجل التغييرات (log)
        int result = set_value(filepath, home, argv[2], argv[3], argv[4]);
        if (result == 0) {
            printf(KGRN "✔ Success:" KWHT " Value was set successfully.\n" KRESET);
        }
        return result;
    } else if (strcmp(argv[1], "get") == 0 && argc == 4) {
        // دالة get تطبع القيمة بنفسها عند النجاح، لذا لا تحتاج رسالة إضافية
        return get_value(filepath, argv[2], argv[3]);
    } else if (strcmp(argv[1], "delete") == 0 && argc == 4) {
        // تم تمرير متغير 'home' لإنشاء سجل التغييرات (log)
        int result = delete_value(filepath, home, argv[2], argv[3]);
        if (result == 0) {
            printf(KGRN "✔ Success:" KWHT " Value was deleted successfully.\n" KRESET);
        }
        return result;
    } else {
        print_help();
        return 1;
    }
    // --- نهاية التعديلات ---
}
